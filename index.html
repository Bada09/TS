Query: MTBF por Máquina

SELECT machine_id,
  AVG(gap_h) AS mtbf_horas
FROM (
  SELECT machine_id,
    EXTRACT(epoch FROM time - LAG(time)
    OVER (PARTITION BY machine_id ORDER BY time))/3600 gap_h
  FROM machine_events
  WHERE event_type='alarm'
  AND time > NOW()-'30d'
) sub GROUP BY 1;
